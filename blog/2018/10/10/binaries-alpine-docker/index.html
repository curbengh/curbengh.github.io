Above list is built using the following command:

script.shsource
#!/bin/sh

IFS=":"
for i in $PATH
do
  if [ -d "$i" ]; then
    # -L follows symlinks
    find -L "$i" -maxdepth 1 -executable -type f -exec basename {} \; 2> /dev/null || :
  fi
done
